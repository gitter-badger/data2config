# 数据描述


CSV {
    string      name             // 文件名 xxx.csv
    string      trimmedName      // 文件名 xxx
    string      fileName         // 文件名 d:/xxx.csv
    <<value>>   data          // 数据 [line][column]
}

# class
Manage {
    <string> templateNames         // 管理类的模板名
    <string> clsTemplateName       // 类的默认模板名

private:
    <Template> _templates
    <Template> _clsTemplates
}
Var {
    string      name            // 属性名
    string      value           // 属性值  仅当Instance下有效
    string      type            // 属性类型
}

################## for class manager template ###################

# 表示描述文件中的一个类，
# 模板变量 classes<Class> 表示class数组
# 模板变量 classmap<name, Class>
Class {
    CSV         csv              // 对应的csv
    string      templateNames    // 对应的模板名
    string      name             // 类名
    <string>    indexNames       // 索引属性名列表
    <str|num>   indexValues      // 索引属性值列表
    bool        isMap            // 是否索引为key
    <Var>       vars             // 属性列表
    
private:
    Template _templates         // 对应的模板
}


################### for csv file template #####################


# 表示csv里的一行数据
# 模板变量 instances<Instance> 数组
# 模板变量 instancemap<instanceKey, Instance> 
Instance {
    Class       class           // 对应的类
    string      key             // 实力的key变量 (仅当是map形式的时候)
    string      value           // 实例的key值   (仅当是map形式的时候)
    <Var>       vars            // 属性列表 等同于 class.vars var.type 有效
}
